Cheatsheet Discord bot that I wrote in Python 3

Lastest Updated on Apr, 18 2020 5:18PM

   Discord is the best for chatting with my friend and make friends, it's comfortable with gaming so, this is some code that I wrote. Let's see this.

@client.event

  • on_ready()
    async def on_ready():
       print('Bot is ready.')
  • on_member_join()
    async def on_member_join(member):
       print(f'{member} has joined a server.')
  • on_member_remove()
    async def on_member_remove(member):
      print(f'{member} has left a server.')
  • on_reaction_add(reaction, user), on_reaction_remove(reaction, user)
    channel = reaction.message.channel
    await channel.send(f"{user.name} has added/removed {reaction.emoji} to the message: {reaction.message.content}")

@client.command()

  • add(), sub(), mul(), div()
    async def add(ctx, a: int, b: int):
      await ctx.send(a+b) //you can change to sub, mul or div up to you.
  • help() // you must client.remove_command('help') before type this
    channel = ctx.message.channel embed = discord.Embed(title="Help", description="Help the absolute life.", color=0xeee657)
      embed.add_field(name=".ping", value="show the latency.", inline=False)
      embed.add_field(name=".ask [Question]", value="Ask question to the bot.", inline=False)
      await channel.send(embed=embed)
